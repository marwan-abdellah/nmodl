28 /**
29  * \class RenameVisitor
30  * \brief `Blindly` rename given variable to new name
31  *
32  * During inlining related passes we have to rename variables
33  * to avoid name conflicts. This pass "blindly" rename any given
34  * variable to new name. The error handling / legality checks are
35  * supposed to be done by other higher level passes. For example,
36  * local renaming pass should be done from inner-most block to top
37  * level block;
38  *
39  * \todo Add log/warning messages.
40  */
41 class RenameVisitor: public AstVisitor {
42  private:
43  /// variable to rename
44  std::string var_name;
45 
46  /// new name
47  std::string new_var_name;
48 
49  // rename verbatim blocks as well
50  bool rename_verbatim = true;
51 
52  public:
53  RenameVisitor() = default;
54 
55  RenameVisitor(std::string old_name, std::string new_name)
56  : var_name(std::move(old_name))
57  , new_var_name(std::move(new_name)) {}
58 
59  void set(std::string old_name, std::string new_name) {
60  var_name = std::move(old_name);
61  new_var_name = std::move(new_name);
62  }
63 
64  void enable_verbatim(bool state) {

66  }
67 
68  void visit_name(ast::Name& node) override;
