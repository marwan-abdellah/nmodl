26 /**
27  * How symbols are stored in NEURON? See notes written in markdown file.
28  *
29  * Some variables get printed by iterating over symbol table in mod2c.
30  * The example of this is thread variables (and also ions?). In this
31  * case we must have to arrange order if we are going keep compatibility
32  * with NEURON.
33  *
34  * Suppose there are three global variables: bcd, abc, abd, abe
35  * They will be in the 'a' bucket in order:
36  * abe, abd, abc
37  * and in 'b' bucket
38  * bcd
39  * So when we print thread variables, we first have to sort in the opposite
40  * order in which they come and then again order by first character in increasing
41  * order.
42  *
43  * Note that variables in double array do not need this transformation
44  * and it seems like they should just follow definition order.
45  */
46 void CodegenHelperVisitor::sort_with_mod2c_symbol_order(std::vector<SymbolType>& symbols) const {
47  /// first sort by global id to get in reverse order
48  std::sort(symbols.begin(),
49  symbols.end(),
50  [](const SymbolType& first, const SymbolType& second) -> bool {
51  return first->get_id() > second->get_id();
52  });
53 
54  /// now order by name (to be same as neuron's bucket)
55  std::sort(symbols.begin(),
56  symbols.end(),
57  [](const SymbolType& first, const SymbolType& second) -> bool {
58  return first->get_name()[0] < second->get_name()[0];
59  });
60 }

63 /**
64  * Find all ions used in mod file
65  */
66 void CodegenHelperVisitor::find_ion_variables() {
67  /// name of the ions used
68  auto ion_vars = psymtab->get_variables_with_properties(NmodlType::useion);
69  /// read variables from all ions
70  auto read_ion_vars = psymtab->get_variables_with_properties(NmodlType::read_ion_var);
71  /// write variables from all ions
72  auto write_ion_vars = psymtab->get_variables_with_properties(NmodlType::write_ion_var);
73 
74  /**
75  * Check if given variable belongs to given ion.
76  * For example, eca belongs to ca ion, nai belongs to na ion.
77  * We just check if we exclude first/last char, if that is ion name.
78  */
79  auto ion_variable = [](const std::string& var, const std::string& ion) -> bool {
80  auto len = var.size() - 1;
81  return (var.substr(1, len) == ion || var.substr(0, len) == ion);
82  };
83 
84  /// iterate over all ion types and construct the Ion objects
85  for (auto& ion_var: ion_vars) {
86  auto ion_name = ion_var->get_name();
87  Ion ion(ion_name);
88  for (auto& read_var: read_ion_vars) {
89  auto var = read_var->get_name();
90  if (ion_variable(var, ion_name)) {
91  ion.reads.push_back(var);
92  }
93  }
94  for (auto& write_var: write_ion_vars) {
95  auto varname = write_var->get_name();
96  if (ion_variable(varname, ion_name)) {
97  ion.writes.push_back(varname);
98  if (ion.is_intra_cell_conc(varname) || ion.is_extra_cell_conc(varname)) {
99  ion.need_style = true;
100  info.write_concentration = true;
101  }
102  }
103  }
104  info.ions.push_back(std::move(ion));
105  }
106 
107  /// once ions are populated, we can find all currents
108  auto vars = psymtab->get_variables_with_properties(NmodlType::nonspecific_cur_var);
109  for (auto& var: vars) {
110  info.currents.push_back(var->get_name());
111  }
112  vars = psymtab->get_variables_with_properties(NmodlType::electrode_cur_var);
113  for (auto& var: vars) {
114  info.currents.push_back(var->get_name());
115  }
116  for (auto& ion: info.ions) {
117  for (auto& var: ion.writes) {
118  if (ion.is_ionic_current(var)) {
119  info.currents.push_back(var);
120  }
121  }
122  }
123 
124  /// check if worte_conc(...) will be needed
125  for (const auto& ion: info.ions) {
126  for (const auto& var: ion.writes) {
127  if (!ion.is_ionic_current(var) && !ion.is_rev_potential(var)) {
128  info.require_wrote_conc = true;
129  }
130  }
131  }
132 }

135 /**
136  * Find non-range variables i.e. ones that are not belong to per instance allocation
137  *
138  * Certain variables like pointers, global, parameters are not necessary to be per
139  * instance variables. NEURON apply certain rules to determine which variables become
140  * thread, static or global variables. Here we construct those variables.
141  */
142 void CodegenHelperVisitor::find_non_range_variables() {
143  /**
144  * Top local variables are local variables appear in global scope. All local
145  * variables in program symbol table are in global scope.
146  */
147  info.constant_variables = psymtab->get_variables_with_properties(NmodlType::constant_var);
148  info.top_local_variables = psymtab->get_variables_with_properties(NmodlType::local_var);
149 
150  /**
151  * All global variables remain global if mod file is not marked thread safe.
152  * Otherwise, global variables written at least once gets promoted to thread variables.
153  */
154 
155  std::string variables;
156 
157  auto vars = psymtab->get_variables_with_properties(NmodlType::global_var);
158  for (auto& var: vars) {
159  if (info.thread_safe && var->get_write_count() > 0) {
160  var->mark_thread_safe();
161  info.thread_variables.push_back(var);
162  info.thread_var_data_size += var->get_length();
163  variables += " " + var->get_name();
164  } else {
165  info.global_variables.push_back(var);
166  }
167  }
168 
169  /**
170  * If parameter is not a range and used only as read variable then it becomes global
171  * variable. To qualify it as thread variable it must be be written at least once and
172  * mod file must be marked as thread safe.
173  * To exclusively get parameters only, we exclude all other variables (in without)
174  * and then sort them with neuron/mod2c order.
175  */
176  // clang-format off
177  auto with = NmodlType::param_assign;
178  auto without = NmodlType::range_var
179  | NmodlType::assigned_definition
180  | NmodlType::global_var
181  | NmodlType::pointer_var
182  | NmodlType::bbcore_pointer_var
183  | NmodlType::read_ion_var
184  | NmodlType::write_ion_var;
185  // clang-format on
186  vars = psymtab->get_variables(with, without);
187  for (auto& var: vars) {
188  // some variables like area and diam are declared in parameter
189  // block but they are not global
190  if (var->get_name() == naming::DIAM_VARIABLE || var->get_name() == naming::AREA_VARIABLE ||
191  var->has_any_property(NmodlType::extern_neuron_variable)) {
192  continue;
193  }
194 
195  // if model is thread safe and if parameter is being written then
196  // those variables should be promoted to thread safe variable
197  if (info.thread_safe && var->get_write_count() > 0) {
198  var->mark_thread_safe();
199  info.thread_variables.push_back(var);
200  info.thread_var_data_size += var->get_length();
201  } else {
202  info.global_variables.push_back(var);
203  }
204  }
205  sort_with_mod2c_symbol_order(info.thread_variables);
206 
207  /**
208  * \todo Below we calculate thread related id and sizes. This will
209  * need to do from global analysis pass as here we are handling
210  * top local variables, global variables, derivimplicit method.
211  * There might be more use cases with other solver methods.
212  */
213 
214  /**
215  * If derivimplicit is used, then first three thread ids get assigned to:
216  * 1st thread is used for: deriv_advance
217  * 2nd thread is used for: dith
218  * 3rd thread is used for: newtonspace
219  *
220  * slist and dlist represent the offsets for prime variables used. For
221  * euler or derivimplicit methods its always first number.
222  */
223  if (info.derivimplicit_used) {
224  info.derivimplicit_var_thread_id = 0;
225  info.thread_data_index = 3;
226  info.derivimplicit_list_num = 1;
227  info.thread_callback_register = true;
228  }
229 
230  /// next thread id is allocated for top local variables
231  if (info.vectorize && !info.top_local_variables.empty()) {
232  info.top_local_thread_id = info.thread_data_index++;
233  info.thread_callback_register = true;
234  }
235 
236  /// next thread id is allocated for thread promoted variables
237  if (info.vectorize && !info.thread_variables.empty()) {
238  info.thread_var_thread_id = info.thread_data_index++;
239  info.thread_callback_register = true;
240  }
241 
242  /// find total size of local variables in global scope
243  for (auto& var: info.top_local_variables) {
244  info.top_local_thread_size += var->get_length();
245  }
246 
247  /// find number of prime variables and total size
248  auto primes = psymtab->get_variables_with_properties(NmodlType::prime_name);
249  info.num_primes = primes.size();
250  for (auto& variable: primes) {
251  info.primes_size += variable->get_length();
252  }
253 
254  /// find pointer or bbcore pointer variables
255  // clang-format off
256  auto properties = NmodlType::pointer_var
257  | NmodlType::bbcore_pointer_var;
258  // clang-format on
259  info.pointer_variables = psymtab->get_variables_with_properties(properties);
260 
261  // find special variables like diam, area
262  // clang-format off
263  properties = NmodlType::assigned_definition
264  | NmodlType::param_assign;
265  vars = psymtab->get_variables_with_properties(properties);
266  for (auto& var : vars) {
267  if (var->get_name() == naming::AREA_VARIABLE) {
268  info.area_used = true;
269  }
270  if (var->get_name() == naming::DIAM_VARIABLE) {
271  info.diam_used = true;
272  }
273  }
274  // clang-format on
275 }

277 /**
278  * Find range variables i.e. ones that are belong to per instance allocation
279  *
280  * In order to be compatible with NEURON, we need to print range variables in
281  * certain order. For example, range variables which are parameters comes first.
282  * Also, there is difference between declaration order vs. definition order. For
283  * example, POINTER variable in NEURON block is just declaration and doesn't
284  * determine the order in which they will get printed. Below we query symbol table
285  * and order all instance variables into certain order.
286  */
287 void CodegenHelperVisitor::find_range_variables() {
288  /// comparator to decide the order based on definition
289  auto comparator = [](const SymbolType& first, const SymbolType& second) -> bool {
290  return first->get_definition_order() < second->get_definition_order();
291  };
292 
293  /**
294  * First come parameters which are range variables.
295  */
296  // clang-format off
297  auto with = NmodlType::range_var
298  | NmodlType::param_assign;
299  auto without = NmodlType::global_var
300  | NmodlType::pointer_var
301  | NmodlType::bbcore_pointer_var
302  | NmodlType::state_var;
303  // clang-format on
304  info.range_parameter_vars = psymtab->get_variables(with, without);
305  std::sort(info.range_parameter_vars.begin(), info.range_parameter_vars.end(), comparator);
306 
307  /**
308  * Second come assigned variables which are range variables.
309  */
310  // clang-format off
311  with = NmodlType::range_var
312  | NmodlType::assigned_definition;
313  without = NmodlType::global_var
314  | NmodlType::pointer_var
315  | NmodlType::bbcore_pointer_var
316  | NmodlType::state_var
317  | NmodlType::param_assign;
318  // clang-format on
319  info.range_assigned_vars = psymtab->get_variables(with, without);
320  std::sort(info.range_assigned_vars.begin(), info.range_assigned_vars.end(), comparator);
321 
322  /**
323  * Third come state variables. All state variables are kind of range by default.
324  * Note that some mod files like CaDynamics_E2.mod use cai as state variable
325  * and those are not considered as range+state variables while printing instance
326  * variables. Such read/write ion variables are assigned variables and hence they
327  * will be printed at laster stage.
328  * \todo Need to validate with more models and mod2c details.
329  */
330  // clang-format off
331  with = NmodlType::state_var;
332  without = NmodlType::global_var
333  | NmodlType::pointer_var
334  | NmodlType::bbcore_pointer_var
335  | NmodlType::read_ion_var
336  | NmodlType::write_ion_var;
337  // clang-format on
338  info.state_vars = psymtab->get_variables(with, without);
339  std::sort(info.state_vars.begin(), info.state_vars.end(), comparator);
340 
341  /**
342  * Remaining variables are:
343  * - all assigned variables without range
344  * - read ion variables which appear in parameter or assigned block
345  * - state variables which are not range but with ion variable of read/write type
346  */
347 
348  /**
349  * first get assigned definition without read ion variables
350  */
351  // clang-format off
352  with = NmodlType::assigned_definition;
353  without = NmodlType::global_var
354  | NmodlType::pointer_var
355  | NmodlType::bbcore_pointer_var
356  | NmodlType::state_var
357  | NmodlType::range_var
358  | NmodlType::extern_neuron_variable
359  | NmodlType::read_ion_var;
360  // clang-format on
361  info.assigned_vars = psymtab->get_variables(with, without);
362 
363  /**
364  * Now just use read-ion variables because every read-ion variable
365  * must be part of either assigned or parameter block. Otherwise code is not
366  * compiled anyway.
367  */
368  // clang-format off
369  with = NmodlType::read_ion_var;
370  without = NmodlType::global_var
371  | NmodlType::pointer_var
372  | NmodlType::bbcore_pointer_var
373  | NmodlType::state_var
374  | NmodlType::range_var
375  | NmodlType::extern_neuron_variable;
376  // clang-format on
377  auto variables = psymtab->get_variables(with, without);
378  info.assigned_vars.insert(info.assigned_vars.end(), variables.begin(), variables.end());
379 
380  /*
381  * We want to have state variables which are read or write ion variables.
382  * This needs to be separated from other state variables because mod2c
383  * treat them separately for ordering.
384  */
385  // clang-format off
386  with = NmodlType::state_var;
387  without = NmodlType::global_var
388  | NmodlType::pointer_var
389  | NmodlType::bbcore_pointer_var
390  | NmodlType::range_var
391  | NmodlType::extern_neuron_variable;
392  // clang-format on
393  variables = psymtab->get_variables(with, without);
394  for (auto& variable: variables) {
395  // clang-format off
396  auto properties = NmodlType::read_ion_var
397  | NmodlType::write_ion_var;
398  // clang-format on
399  if (variable->has_any_property(properties)) {
400  info.ion_state_vars.push_back(variable);
401  info.assigned_vars.push_back(variable);
402  }
403  }
404 }

528 /**
529  * Visit statement block and find prime symbols appear in derivative block
530  *
531  * Equation statements in derivative block has prime on the lhs. The order
532  * of primes could be different that declaration state block. Also, not all
533  * state variables need to appear in equation block. In this case, we want
534  * to find out the the primes in the order of equation definition. This is
535  * just to keep the same order as neuron implementation.
536  *
537  * The primes are already solved and replaced by Dstate or name. And hence
538  * we need to check if the lhs variable is derived from prime name. If it's
539  * Dstate then we have to lookup state to find out corresponding symbol. This
540  * is because prime_variables_by_order should contain state variable name and
541  * not the one replaced by solver pass.
542  */
543 void CodegenHelperVisitor::visit_statement_block(ast::StatementBlock& node) {
544  const auto& statements = node.get_statements();
545  for (auto& statement: statements) {
546  statement->accept(*this);
547  if (under_derivative_block && assign_lhs &&
548  (assign_lhs->is_name() || assign_lhs->is_var_name())) {
549  auto name = assign_lhs->get_node_name();
550  auto symbol = psymtab->lookup(name);
551  if (symbol != nullptr) {
552  auto is_prime = symbol->has_any_property(NmodlType::prime_name);
553  auto from_state = symbol->has_any_status(Status::from_state);
554  if (is_prime || from_state) {
555  if (from_state) {
556  symbol = psymtab->lookup(name.substr(1, name.size()));
557  }
558  info.prime_variables_by_order.push_back(symbol);
559  info.num_equations++;
560  }
561  }
562  }
563  assign_lhs = nullptr;
564  }
565 }

608 void CodegenHelperVisitor::visit_program(ast::Program& node) {
609  psymtab = node.get_symbol_table();
610  auto blocks = node.get_blocks();
611  for (auto& block: blocks) {
612  info.top_blocks.push_back(block.get());
613  if (block->is_verbatim()) {
614  info.top_verbatim_blocks.push_back(block.get());
615  }
616  }
617  node.visit_children(*this);
618  find_range_variables();
619  find_non_range_variables();
620  find_ion_variables();
621  find_table_variables();
622 }
