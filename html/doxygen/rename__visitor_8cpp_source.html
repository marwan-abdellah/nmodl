35 /**
36  * Parse verbatim blocks and rename variable if it is used.
37  */
38 void RenameVisitor::visit_verbatim(ast::Verbatim& node) {
39  if (!rename_verbatim) {
40  return;
41  }
42 
43  const auto& statement = node.get_statement();
44  auto text = statement->eval();
45  parser::CDriver driver;
46 
47  driver.scan_string(text);
48  auto tokens = driver.all_tokens();
49 
50  std::string result;
51  for (auto& token: tokens) {
52  if (token == var_name) {
53  result += new_var_name;
54  } else {
55  result += token;
56  }
57  }
58  statement->set(result);
59 }
